> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为图例设置补齐 `legend.box.just` 对齐方式参数，并让分类图例刷新逻辑统一控制“图例设置”tab 显隐和列数、刻度数展示规则。

```js
function createLegendMeta() {
  return {
    typeSet: new Set(),
    valueTypeSet: new Set(),
    continuousValueTypeSet: new Set()
  }
}

function normalizeNumberList(valueSet) {
  return Array.from(valueSet)
    .filter(value => value !== null && value !== undefined)
    .sort((left, right) => left - right)
}

function resolvePreferredValueType(valueTypeList) {
  if (valueTypeList.includes(COLOR)) {
    return COLOR
  }
  return valueTypeList.length > 0 ? valueTypeList[0] : null
}

function applyLegendMeta(classifyLegendItem, legendMeta) {
  const valueTypeList = normalizeNumberList(legendMeta.valueTypeSet)
  const continuousValueTypeList = normalizeNumberList(
    legendMeta.continuousValueTypeSet
  )

  classifyLegendItem.valueType = resolvePreferredValueType(valueTypeList)
  classifyLegendItem.valueTypeList = valueTypeList
  classifyLegendItem.continuousValueTypeList = continuousValueTypeList
}

function setLegendSettingsGroupVisible(commonParams, classifyLegendSetting) {
  const legendSettingsGroup = localFindGroupFromTopParamSettingsListByName(
    "图例设置",
    commonParams
  )

  if (legendSettingsGroup) {
    legendSettingsGroup.visible = CommonUtils.isNotEmptyArray(
      classifyLegendSetting.value
    )
  }
}
```

```js
function createClassifyLegendItem(by, type, legendMeta) {
  const defaultClassifyLegend = window.defaultClassifyLegend

  if (!defaultClassifyLegend) {
    return null
  }

  const legend = CommonUtils.deepClone(defaultClassifyLegend)

  if (!legend.title) {
    legend.title = {}
  }
  if (!legend.by) {
    legend.by = {}
  }

  legend.title.value = by
  legend.by.value = by

  const classifyLegendItem = {
    type: type,
    legend: legend
  }

  applyLegendMeta(classifyLegendItem, legendMeta)

  return classifyLegendItem
}
```

```js
function syncClassifyLegendList(classifyLegendSetting, byMetaMap) {
  const classifyLegendList = []
  const retainedBySet = new Set()
  const oldClassifyLegendList = CommonUtils.isNotEmptyArray(classifyLegendSetting.value)
    ? classifyLegendSetting.value
    : []

  for (let index = 0; index < oldClassifyLegendList.length; index++) {
    const classifyLegendItem = oldClassifyLegendList[index]
    const by = resolveLegendItemBy(classifyLegendItem)

    if (!by || !byMetaMap.has(by) || retainedBySet.has(by)) {
      continue
    }

    const legendMeta = byMetaMap.get(by)
    classifyLegendItem.type = resolveLegendType(legendMeta.typeSet)
    applyLegendMeta(classifyLegendItem, legendMeta)
    classifyLegendList.push(classifyLegendItem)
    retainedBySet.add(by)
  }

  byMetaMap.forEach((legendMeta, by) => {
    if (retainedBySet.has(by)) {
      return
    }

    const classifyLegendItem = createClassifyLegendItem(
      by,
      resolveLegendType(legendMeta.typeSet),
      legendMeta
    )

    if (classifyLegendItem) {
      classifyLegendList.push(classifyLegendItem)
      retainedBySet.add(by)
    }
  })

  classifyLegendSetting.value = classifyLegendList
}
```

将函数内 `byTypeMap` 相关逻辑替换为：

```js
  // 2. 收集所有分类参数中的 by 字段、分类类型和视觉属性类型。
  // 使用 Map 来存储 by -> 元数据集合，保证同一 by 被多个分类参数复用时可以增量聚合。
  const byMetaMap = new Map()

  if (CommonUtils.isNotEmptyArray(commonParams.topParamSettingsList)) {
    // 遍历所有分组（moduleType === -1）
    for (let i = 0; i < commonParams.topParamSettingsList.length; i++) {
      const group = commonParams.topParamSettingsList[i]

      if (group.moduleType === -1 && CommonUtils.isNotEmptyArray(group.paramSettingsList)) {
        // 遍历分组中的所有参数
        for (let j = 0; j < group.paramSettingsList.length; j++) {
          const param = group.paramSettingsList[j]

          // 找到分类参数（moduleType === 7）
          if (param.moduleType === 7) {
            // 检查 classifyItemList 是否存在且有内容
            if (
              CommonUtils.isNotEmptyArray(param.classifyItemList) &&
              param.classifyItemList.length > 0
            ) {
              const classifyItem = param.classifyItemList[0]
              // 分类子项关闭图例时，不参与 legendList 汇总；否则旧图例会在刷新时被重新保留或创建。
              if (classifyItem.showLegend === false) {
                continue
              }

              const by = classifyItem.by
              const type = classifyItem.type // 1-离散型, 2-连续型
              const valueType = classifyItem.valueType

              // 过滤掉空的 by 字段
              if (CommonUtils.isNotEmptyString(by) && type) {
                if (!byMetaMap.has(by)) {
                  byMetaMap.set(by, createLegendMeta())
                }

                const legendMeta = byMetaMap.get(by)
                legendMeta.typeSet.add(type)

                if (valueType !== null && valueType !== undefined) {
                  legendMeta.valueTypeSet.add(valueType)
                  if (type === CLASSIFY_CONTINUES) {
                    legendMeta.continuousValueTypeSet.add(valueType)
                  }
                }
              }
            }
          }
        }
      }
    }
  }

  // 3. 根据 byMetaMap 增量同步 classifyLegendSetting.value
  syncClassifyLegendList(classifyLegendSetting, byMetaMap)

  // 4. 图例数量为 0 时隐藏固定名称的“图例设置”分组；找不到该分组时不做降级处理。
  setLegendSettingsGroupVisible(commonParams, classifyLegendSetting)
```

```js
    // 判断图例项是否包含连续颜色映射。连续颜色使用刻度数，连续大小和粗细仍使用列数。
    isContinuousColorLegend(classifyLegendItem) {
      if (!classifyLegendItem) {
        return false
      }

      if (Array.isArray(classifyLegendItem.continuousValueTypeList)) {
        return classifyLegendItem.continuousValueTypeList.includes(1)
      }

      return classifyLegendItem.type === 2 && classifyLegendItem.valueType === 1
    },

    // 列数用于离散图例、混合图例，以及连续非颜色图例。
    shouldShowNcol(classifyLegendItem) {
      if (!classifyLegendItem) {
        return false
      }

      if (classifyLegendItem.type === 1 || classifyLegendItem.type === 3) {
        return true
      }

      if (classifyLegendItem.type === 2) {
        return !this.isContinuousColorLegend(classifyLegendItem)
      }

      return true
    },

    // 刻度数只用于连续颜色图例；混合图例中包含连续颜色时也需要保留刻度数。
    shouldShowNticks(classifyLegendItem) {
      return this.isContinuousColorLegend(classifyLegendItem)
    },
```
