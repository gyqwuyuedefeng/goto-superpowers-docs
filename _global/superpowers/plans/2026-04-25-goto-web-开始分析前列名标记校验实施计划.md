- [ ] **Step 3: 在 `draw.vue` 中实现最小校验逻辑**

Add this shape inside `methods` in `goto-web/src/views/goto/plot/type/draw.vue`:

```js
handleSubmit() {
  if (!this.validateRequiredFileMarks()) {
    return
  }
  this.doSubmit()
},

validateRequiredFileMarks() {
  const fileSettings = (this.plot && this.plot.userDraw && this.plot.userDraw.fileSettings) || []
  if (!this.CommonUtils.isNotEmptyArray(fileSettings)) {
    return true
  }

  for (const fileSetting of fileSettings) {
    const fileMarkContainerList = (fileSetting && fileSetting.fileMarkContainerList) || []
    const requiredContainers = fileMarkContainerList.filter(
      container => container && container.type === 'SELECT_COLUMN' && container.required === true
    )

    if (!this.CommonUtils.isNotEmptyArray(requiredContainers)) {
      continue
    }

    const headerInfoList = (fileSetting && fileSetting.headerInfoList) || []

    for (const container of requiredContainers) {
      const marked = headerInfoList.some(
        headerInfo => this.BitwiseUtils.getBit(headerInfo.status || 0, container.positionParam) === 1
      )

      if (!marked) {
        this.ToastificationNotice.notifyError(this, '请先标记列名')
        return false
      }
    }
  }

  return true
}
```

Expected: 只在“开始分析”入口增加统一拦截，不改动 `fileMark/fileSetting.vue` 内部保存逻辑。

```java
private boolean hasMissingRequiredFileMark(List<ChartUploadFileSettings> fileSettings) {
    if (CollectionUtils.isEmpty(fileSettings)) {
        return false;
    }

    for (ChartUploadFileSettings fileSetting : fileSettings) {
        List<FileMarkContainer> containers = fileSetting.getFileMarkContainerList();
        if (CollectionUtils.isEmpty(containers)) {
            continue;
        }

        List<FileMarkContainer> requiredContainers = containers.stream()
                .filter(container -> container != null
                        && container.getType() == FileMarkContainerType.SELECT_COLUMN
                        && Boolean.TRUE.equals(container.getRequired()))
                .toList();

        if (CollectionUtils.isEmpty(requiredContainers)) {
            continue;
        }

        if (CollectionUtils.isEmpty(fileSetting.getHeaderInfoList())) {
            return true;
        }

        for (FileMarkContainer container : requiredContainers) {
            boolean marked = fileSetting.getHeaderInfoList().stream()
                    .filter(Objects::nonNull)
                    .anyMatch(headerInfo -> BitwiseUtils.active(
                            BitwiseUtils.getBit(headerInfo.getStatus(), container.getPositionParam())));
            if (!marked) {
                return true;
            }
        }
    }

    return false;
}
```

Expected: 校验放在 `UserDrawUtils.userDrawUpdate(...)` 之前；失败时直接返回，不触发状态更新或消息发送。
