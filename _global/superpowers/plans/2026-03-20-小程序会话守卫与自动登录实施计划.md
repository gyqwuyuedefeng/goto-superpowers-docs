**Goal:** 为 `goto-wechat` 小程序实现首页会话守卫、静默自动登录、统一的 `401/403` 会话恢复，以及扫码登录后的首页收口。

Create `goto-wechat/utils/session.js` with a factory + singleton shape that keeps token persistence centralized:

```js
const { authApi } = require("./api");
const { configManager } = require("./config");

function normalizeToken(rawToken) {
  if (!rawToken) return "";
  return String(rawToken).replace(/^Bearer\s+/i, "");
}

function createSessionManager(deps = {}) {
  const wxApi = deps.wxApi || wx;
  const auth = deps.authApi || authApi;
  const tokenStorageKey =
    deps.tokenStorageKey || configManager.get("TOKEN_STORAGE_KEY");

  let loginPromise = null;
  let recoveryPromise = null;
  let entryState = null;

  return {
    getToken() {
      return wxApi.getStorageSync(tokenStorageKey) || "";
    },
    saveAuthResult(authInfo) {
      const token = normalizeToken(authInfo && authInfo.token);
      if (token) {
        wxApi.setStorageSync(tokenStorageKey, token);
      }
      return token;
    },
    clearSession() {
      wxApi.removeStorageSync(tokenStorageKey);
    },
    setEntryState(nextState) {
      entryState = nextState;
    },
    consumeEntryState() {
      const current = entryState;
      entryState = null;
      return current;
    },
    async silentLogin() {
      if (loginPromise) {
        return loginPromise;
      }

      loginPromise = (async () => {
        const code = await new Promise((resolve, reject) => {
          wxApi.login({
            success: ({ code }) =>
              code ? resolve(code) : reject(new Error("missing code")),
            fail: reject,
          });
        });
        const response = await auth.miniProgramLogin(code, {
          skipAuthRecovery: true,
        });
        this.saveAuthResult(response.data);
        return response;
      })();

      try {
        return await loginPromise;
      } finally {
        loginPromise = null;
      }
    },
    async ensureSession() {
      try {
        if (this.getToken()) {
          await auth.getInfo({ skipAuthRecovery: true });
          return { status: "ready" };
        }
        await this.silentLogin();
        await auth.getInfo({ skipAuthRecovery: true });
        return { status: "ready" };
      } catch (error) {
        if (error && (error.statusCode === 401 || error.statusCode === 403)) {
          this.clearSession();
          await this.silentLogin();
          await auth.getInfo({ skipAuthRecovery: true });
          return { status: "ready" };
        }
        throw error;
      }
    },
    getRecoveryPromise() {
      return recoveryPromise;
    },
    setRecoveryPromise(promise) {
      recoveryPromise = promise;
    },
  };
}

const sessionManager = createSessionManager();

module.exports = {
  normalizeToken,
  createSessionManager,
  sessionManager,
};
```

```js
let sessionFlowPromise = null;

function runSessionFlow(work) {
  if (sessionFlowPromise) {
    return sessionFlowPromise;
  }

  sessionFlowPromise = (async () => {
    try {
      return await work();
    } finally {
      sessionFlowPromise = null;
    }
  })();

  return sessionFlowPromise;
}

async ensureSession() {
  return runSessionFlow(async () => {
    try {
      if (this.getToken()) {
        await auth.getInfo({ skipAuthRecovery: true, showLoading: false });
        return { status: "ready" };
      }
      await this.silentLogin();
      await auth.getInfo({ skipAuthRecovery: true, showLoading: false });
      return { status: "ready" };
    } catch (error) {
      if (error && (error.statusCode === 401 || error.statusCode === 403)) {
        this.clearSession();
        await this.silentLogin();
        await auth.getInfo({ skipAuthRecovery: true, showLoading: false });
        return { status: "ready" };
      }
      throw error;
    }
  });
}

async handleAuthFailure(error) {
  return runSessionFlow(async () => {
    this.clearSession();
    try {
      await this.silentLogin();
      await auth.getInfo({ skipAuthRecovery: true, showLoading: false });
      return { recovered: true };
    } catch (recoveryError) {
      this.setEntryState({ mode: "failed" });
      wxApi.reLaunch({ url: "/pages/index/index" });
      throw recoveryError;
    }
  });
}
```
