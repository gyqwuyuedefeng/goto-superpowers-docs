> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Tech Stack:** Java 17, Spring Boot, JPA, JUnit 5, AssertJ, Mockito, Apache Commons IO

  package com.freedom.quartz.task;

  import com.freedom.bean.ComponentBean;
  import com.freedom.domain.UserDrawSimple;
  import com.freedom.util.UserDrawUtils;
  import lombok.RequiredArgsConstructor;
  import lombok.extern.slf4j.Slf4j;
  import org.apache.commons.io.FileUtils;
  import org.apache.commons.lang3.StringUtils;
  import org.springframework.stereotype.Service;

  import java.io.File;
  import java.io.IOException;

  /**
   * @author gyq
   * @description 安全删除单条 UserDraw（先目录后数据库）
   */
  @Slf4j
  @Service
  @RequiredArgsConstructor
  public class UserDrawRemoveExecutor {

      private final ComponentBean componentBean;

      /**
       * 安全删除单条 UserDraw。
       * 流程：校验参数 -> 删除目录 -> 删除数据库记录
       *
       * @return true 表示本条删除成功，false 表示失败（已记录日志）
       */
      public boolean execute(UserDrawSimple userDraw) {
          if (userDraw == null || userDraw.getId() == null || StringUtils.isBlank(userDraw.getTmpFolderName())) {
              log.error("[UserDrawRemoveExecutor] 参数校验失败: userDraw={}", userDraw);
              return false;
          }

          Long id = userDraw.getId();
          String tmpFolderName = userDraw.getTmpFolderName();

          // 1. 处理目录
          File folder = UserDrawUtils.obtainTmpFolder(
                  componentBean.getProperties().getConfigProperties(), tmpFolderName);

          if (folder.exists()) {
              try {
                  FileUtils.forceDelete(folder);
              } catch (IOException e) {
                  log.error("[UserDrawRemoveExecutor] 目录删除失败: id={}, tmpFolderName={}", id, tmpFolderName, e);
                  return false;
              }
          }
          // 目录不存在视为目录删除成功，继续删数据库

          // 2. 删除数据库记录
          try {
              componentBean.getRepository().getUserDrawRepository().deleteById(id);
          } catch (Exception e) {
              log.error("[UserDrawRemoveExecutor] 数据库删除失败: id={}, tmpFolderName={}", id, tmpFolderName, e);
              return false;
          }

          return true;
      }
  }
  ```

  package com.freedom.quartz.task;

  import com.freedom.bean.ComponentBean;
  import com.freedom.domain.UserDrawSimple;
  import lombok.RequiredArgsConstructor;
  import lombok.extern.slf4j.Slf4j;
  import org.apache.commons.lang3.ObjectUtils;
  import org.springframework.stereotype.Service;

  import java.util.List;

  /**
   * @author gyq
   * @description 扫描 3 天未使用的 UserDraw 并直接删除（不依赖 del_mark / 延迟队列）
   */
  @Slf4j
  @Service
  @RequiredArgsConstructor
  public class LongTimeUnusedUserDrawRemoveTask {

      private static final int UNUSED_DAYS = 3;

      private final ComponentBean componentBean;
      private final UserDrawRemoveExecutor userDrawRemoveExecutor;

      public void run() {
          log.info("[LongTimeUnusedUserDrawRemoveTask] 任务开始");

          List<UserDrawSimple> list = componentBean.getRepository()
                  .getUserDrawSimpleRepository()
                  .findUnusedProjects(UNUSED_DAYS);

          if (ObjectUtils.isEmpty(list)) {
              log.info("[LongTimeUnusedUserDrawRemoveTask] 无过期数据，任务结束");
              return;
          }

          int total = list.size();
          int success = 0;
          int fail = 0;

          for (UserDrawSimple userDraw : list) {
              boolean ok = userDrawRemoveExecutor.execute(userDraw);
              if (ok) {
                  success++;
              } else {
                  fail++;
              }
          }

          log.info("[LongTimeUnusedUserDrawRemoveTask] 任务结束: 扫描总数={}, 删除成功={}, 删除失败={}",
                  total, success, fail);
      }
  }
  ```
