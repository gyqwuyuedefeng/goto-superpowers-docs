- [ ] **Step 1: 执行迁移脚本**

Run:

```bash
set -euo pipefail

ROOT=/mnt/f/IdeaProjects/goto-software

move_tree() {
  src_dir="$1"
  dest_dir="$2"
  suffix="$3"

  [ -d "$src_dir" ] || return 0
  mkdir -p "$dest_dir"

  find "$src_dir" -maxdepth 1 -type f -print0 | while IFS= read -r -d '' src; do
    base="$(basename "$src")"
    dest="$dest_dir/$base"
    if [ -e "$dest" ]; then
      name="${base%.*}"
      ext=""
      if [ "$name" != "$base" ]; then
        ext=".${base##*.}"
      fi
      dest="$dest_dir/${name}-${suffix}${ext}"
      counter=2
      while [ -e "$dest" ]; do
        dest="$dest_dir/${name}-${suffix}-${counter}${ext}"
        counter=$((counter + 1))
      done
    fi
    mv "$src" "$dest"
  done
}

move_tree "$ROOT/docs/plans" \
  "$ROOT/goto-superpowers-docs/_global/superpowers/plans" \
  "from-docs-plans"

move_tree "$ROOT/docs/superpowers/plans" \
  "$ROOT/goto-superpowers-docs/_global/superpowers/plans" \
  "from-docs-superpowers-plans"

move_tree "$ROOT/docs/superpowers/specs" \
  "$ROOT/goto-superpowers-docs/_global/superpowers/specs" \
  "from-docs-superpowers-specs"

move_tree "$ROOT/goto-web/docs/superpowers/plans" \
  "$ROOT/goto-superpowers-docs/goto-web/superpowers/plans" \
  "from-goto-web-plans"

move_tree "$ROOT/goto-web/docs/superpowers/specs" \
  "$ROOT/goto-superpowers-docs/goto-web/superpowers/specs" \
  "from-goto-web-specs"
```

Expected: Command exits with code 0 and does not print overwrite errors.
