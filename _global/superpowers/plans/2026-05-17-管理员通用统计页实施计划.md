**Tech Stack:** Vue 2、Vue Router、Vuex、Element UI、ECharts 4、Jest、ESLint、GitNexus。

```js
export function normalizeRankList(response) {
  if (Array.isArray(response)) {
    return response
  }
  if (response && Array.isArray(response.data)) {
    return response.data
  }
  return []
}

function toNumber(value) {
  const numberValue = Number(value)
  return Number.isFinite(numberValue) ? numberValue : 0
}

export function buildSummary(rankList) {
  const list = Array.isArray(rankList) ? rankList : []
  const totalDrawCount = list.reduce((sum, item) => {
    return sum + toNumber(item && item.drawCount)
  }, 0)
  const firstItem = list[0]

  return {
    totalDrawCount,
    plotTypeCount: list.length,
    topPlotTypeName: firstItem && firstItem.plotTypeName
      ? firstItem.plotTypeName
      : "暂无数据"
  }
}

export function buildTableRows(rankList) {
  const list = Array.isArray(rankList) ? rankList : []
  return list.map((item, index) => {
    return {
      rank: index + 1,
      plotTypeId: item.plotTypeId,
      plotTypeName: item.plotTypeName || "未命名图类型",
      groupId: item.groupId,
      groupName: item.groupName || "未分组",
      drawCount: toNumber(item.drawCount)
    }
  })
}

export function buildChartData(rankList) {
  const rows = buildTableRows(rankList)
  return {
    names: rows.map(item => item.plotTypeName),
    counts: rows.map(item => item.drawCount)
  }
}
```

```vue
  mounted() {

      if (this.chart) {
        this.chart.resize()
      }
    }, 100)
    window.addEventListener("resize", this.__resizeHandler)
    this.fetchRankList()
  },
  beforeDestroy() {
    window.removeEventListener("resize", this.__resizeHandler)
    if (this.chart) {
      this.chart.dispose()
      this.chart = null
    }
  },
  methods: {
    fetchRankList() {
      this.loading = true
      return popularPlotTypes({
        rangeType: this.rangeType,
        limit: 20
      }).then(response => {
        this.rankList = normalizeRankList(response)
        this.$nextTick(() => {
          this.renderChart()
        })
      }).catch(() => {
        this.rankList = []
        this.$message.error("获取管理员统计数据失败")
        this.$nextTick(() => {
          this.renderChart()
        })
      }).finally(() => {
        this.loading = false
      })
    },
    renderChart() {
      if (!this.rankList.length || !this.$refs.chart) {
        if (this.chart) {
          this.chart.dispose()
          this.chart = null
        }
        return
      }

      if (!this.chart) {
        this.chart = echarts.init(this.$refs.chart, "macarons")
      }

      this.chart.setOption({
        tooltip: {
          trigger: "axis",
          axisPointer: {
            type: "shadow"
          }
        },
        grid: {
          top: 20,
          left: "3%",
          right: "3%",
          bottom: "8%",
          containLabel: true
        },
        xAxis: [
          {
            type: "category",
            data: this.chartData.names,
            axisTick: {
              alignWithLabel: true
            },
            axisLabel: {
              interval: 0,
              rotate: 30
            }
          }
        ],
        yAxis: [
          {
            type: "value",
            minInterval: 1
          }
        ],
        series: [
          {
            name: "绘图次数",
            type: "bar",
            barMaxWidth: 48,
            data: this.chartData.counts
          }
        ]
      }, true)
    }
  }
}
</script>
```
