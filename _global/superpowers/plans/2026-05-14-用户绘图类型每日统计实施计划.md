# 用户绘图类型每日统计实施计划

> **给执行代理：** 必须使用 `superpowers:subagent-driven-development`（推荐）或 `superpowers:executing-plans` 按任务逐步实施本计划。步骤使用复选框（`- [ ]`）语法跟踪。

**目标：** 新增后端数据库每日聚合统计能力，在绘图任务成功提交 Redis Stream 后异步记录用户使用的图类型次数，并提供按时间范围查询热门图类型的后端接口。

**架构：** 在 `goto/common` 新增统计实体、仓储、查询 DTO/VO 和时间范围枚举；在 `goto/system` 新增统计服务、异步服务、控制器，并在 `PlotServiceImpl.customDrawPlot` 成功提交 Redis Stream 后触发异步统计。统计表按 `stat_date + user_id + plot_type_id` 唯一聚合，使用 MySQL upsert 原子自增，查询接口按时间范围汇总 `draw_count`。

**技术栈：** Java 17、Spring Boot 3.3、Spring Data JPA、MySQL native query、Spring `@Async`、JUnit 5、Mockito、Maven。

```java
package com.freedom.model.stat;

import java.time.LocalDate;

/**
 * 用户绘图类型统计查询时间范围。
 */
public enum UserPlotTypeStatRangeType {
    LAST_7_DAYS,
    LAST_30_DAYS,
    LAST_365_DAYS,
    ALL;

    public LocalDate resolveStartDate(LocalDate today) {
        if (today == null) {
            today = LocalDate.now();
        }
        return switch (this) {
            case LAST_7_DAYS -> today.minusDays(6);
            case LAST_30_DAYS -> today.minusDays(29);
            case LAST_365_DAYS -> today.minusDays(364);
            case ALL -> null;
        };
    }

    public static UserPlotTypeStatRangeType safeOf(String value) {
        if (value == null || value.isBlank()) {
            return LAST_7_DAYS;
        }
        try {
            return UserPlotTypeStatRangeType.valueOf(value);
        } catch (IllegalArgumentException e) {
            return LAST_7_DAYS;
        }
    }
}
```

```java
package com.freedom.repository;

import com.freedom.domain.UserPlotTypeDailyStat;
import com.freedom.repository.projection.UserPlotTypePopularProjection;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Modifying;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.util.List;

/**
 * 用户绘图类型每日统计仓储。
 */
public interface UserPlotTypeDailyStatRepository extends JpaRepository<UserPlotTypeDailyStat, Long> {

    @Modifying
    @Transactional
    @Query(value = """
        INSERT INTO app_user_plot_type_daily_stat
            (stat_date, user_id, plot_type_id, draw_count, create_time, update_time)
        VALUES
            (:statDate, :userId, :plotTypeId, 1, NOW(), NOW())
        ON DUPLICATE KEY UPDATE
            draw_count = draw_count + 1,
            update_time = NOW()
        """, nativeQuery = true)
    int incrementDailyStat(
        @Param("statDate") LocalDate statDate,
        @Param("userId") Long userId,
        @Param("plotTypeId") Long plotTypeId
    );

    @Query(value = """
        SELECT
            s.plot_type_id AS plotTypeId,
            p.folder_name AS plotTypeName,
            p.group_id AS groupId,
            g.name AS groupName,
            SUM(s.draw_count) AS drawCount
        FROM app_user_plot_type_daily_stat s
        LEFT JOIN app_plot_type p ON p.id = s.plot_type_id
        LEFT JOIN app_plot_group g ON g.id = p.group_id
        WHERE (:startDate IS NULL OR s.stat_date >= :startDate)
        GROUP BY s.plot_type_id, p.folder_name, p.group_id, g.name
        ORDER BY drawCount DESC, s.plot_type_id ASC
        """, nativeQuery = true)
    List<UserPlotTypePopularProjection> findPopularPlotTypes(
        @Param("startDate") LocalDate startDate,
        Pageable pageable
    );
}
```

```java
/**
 * 用户绘图类型统计服务实现。
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class UserPlotTypeStatServiceImpl implements UserPlotTypeStatService {

    private static final int DEFAULT_LIMIT = 20;
    private static final int MAX_LIMIT = 100;

    private final UserPlotTypeDailyStatRepository repository;

    @Override
    public void incrementDaily(Long userId, Long plotTypeId) {
        if (userId == null || plotTypeId == null) {
            log.warn("跳过用户绘图类型统计，参数为空: userId={}, plotTypeId={}", userId, plotTypeId);
            return;
        }
        LocalDate statDate = LocalDate.now();
        repository.incrementDailyStat(statDate, userId, plotTypeId);
    }

    @Override
    public CommonResponse popularPlotTypes(UserPlotTypePopularQuery query) {
        String rangeType = query == null ? null : query.getRangeType();
        int limit = normalizeLimit(query == null ? null : query.getLimit());
        LocalDate startDate = resolveStartDate(rangeType, LocalDate.now());

        List<UserPlotTypePopularVO> result = repository
            .findPopularPlotTypes(startDate, PageRequest.of(0, limit))
            .stream()
            .map(this::toVO)
            .toList();

        return ResponseUtils.successData(result);
    }

    LocalDate resolveStartDateForTest(String rangeType, LocalDate today) {
        return resolveStartDate(rangeType, today);
    }

    private LocalDate resolveStartDate(String rangeType, LocalDate today) {
        return UserPlotTypeStatRangeType.safeOf(rangeType).resolveStartDate(today);
    }

    private int normalizeLimit(Integer limit) {
        if (limit == null || limit <= 0) {
            return DEFAULT_LIMIT;
        }
        return Math.min(limit, MAX_LIMIT);
    }

    private UserPlotTypePopularVO toVO(UserPlotTypePopularProjection projection) {
        return new UserPlotTypePopularVO(
            projection.getPlotTypeId(),
            projection.getPlotTypeName(),
            projection.getGroupId(),
            projection.getGroupName(),
            projection.getDrawCount()
        );
    }
```
