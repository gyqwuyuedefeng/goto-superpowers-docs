# Rserve 临时目录与通用任务重试闭环修复 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 修正 Rserve 临时目录启动配置和通用 R 连接生命周期，并让 Redis Stream 通用任务在最多 3 次异步重试后稳定进入成功或最终失败终态。

**Architecture:** Rserve 镜像负责在主进程启动前串行创建并导出专用临时目录；`common-thread-r` 在每条连接上执行 `tempdir(check = TRUE)` 与创建/删除探针，并最多重建一次连接。`goto/common` 放开 `FAILED_RETRYABLE → RUNNING` 的原子状态流转，`goto/task` 使用账本中的 `retry_count`、稳定 R 错误类型和配置化上限完成通知与 ACK 闭环。

**Tech Stack:** Bash、Docker、R/Rserve、Java 11/17、Spring Boot 3.3.4、Spring Data JPA、Redis Stream、JUnit 5、Mockito、Maven 3.9.16

## Global Constraints

- 本计划只修改 `goto-rbase-build`、`goto-deploy-backend`、`freedom-public/common-thread-r`、`goto/common` 和 `goto/task`。
- 禁止修改 `gotoTools`、`gotoBase`、`rsession`，`gotoBase::saveTable()` 继续使用 `openxlsx`，Excel 输出内容和表头样式必须保持不变。
- 不新增数据库字段，不执行手工数据库 DML；`stream_job_execution.retry_count` 是异步重试次数唯一事实源。
- 默认允许 3 次异步重试；首次执行和同一次任务内的一次连接自愈不计入这 3 次重试。
- `retry_count=3` 的 `FAILED_RETRYABLE` 任务仍允许执行第 3 次重试；该次再次失败必须进入 `FAILED_FINAL`。历史 `retry_count>3` 的记录不得再次执行业务代码，必须自动终态化并 ACK。
- 连接关闭只调用 R 客户端关闭协议，禁止从 `goto-task` 容器执行 `kill -9 <Rserve PID>`。
- 不批量清理线上长期存活的 Rserve 子进程；只增加可关联的建立、健康检查、关闭日志和上线观测命令。
- 新增或修改的 Java 核心类、方法和复杂状态分支必须写详细中文注释；所有新增或修改的 Java 测试类和测试方法必须包含完整中文测试注释。
- Maven 只能通过 Windows `cmd.exe /c` 调用 `D:\install\work\apache-maven-3.9.16\bin\mvn.cmd`，固定 settings 为 `D:\install\work\apache-maven-3.9.16\conf\settings.xml`，本地仓库为 `E:\repository`，禁止任何 `deploy` 目标。
- Maven 长命令必须用 `tee` 实时显示并保存完整日志，预期成功标志均为 `BUILD SUCCESS`；失败时读取对应完整日志，不得在运行管道末尾使用 `tail`。
- 各仓库当前均存在用户未提交改动。执行前必须使用 `superpowers:using-git-worktrees` 为四个代码仓库分别创建隔离 worktree，严禁覆盖或提交现有工作树中的用户改动。

package com.freedom.util;

import com.freedom.exception.r.RTempDirectoryUnavailableException;
import com.freedom.model.CheckResultMsg;
import lombok.extern.slf4j.Slf4j;
import org.math.R.CustomRserveSession;

/**
 * Rserve 会话临时目录健康检查器。
 *
 * 每条连接在业务命令前调用 tempdir(check = TRUE)，随后验证目录存在、可写，
 * 并真实创建和删除唯一探针文件。检查不会修改业务文件或 gotoTools/gotoBase 行为。
 */
@Slf4j
public class RSessionHealthChecker {

    static final String HEALTH_CHECK_COMMAND =
            "local({"
                    + "temp_dir <- tempdir(check = TRUE);"
                    + "exists <- dir.exists(temp_dir);"
                    + "writable <- exists && file.access(temp_dir, 2L) == 0L;"
                    + "probe <- tempfile('goto-rserve-health-', tmpdir = temp_dir);"
                    + "probe_created <- writable && file.create(probe);"
                    + "if (probe_created) unlink(probe);"
                    + "paste(temp_dir, exists, writable, probe_created, sep = '\\t')"
                    + "})";

    /**
     * 检查指定会话的临时目录，并返回可供日志关联的真实 tempdir 路径。
     *
     * @param session 已建立的 Rserve 会话
     * @return 成功时 data 为 tempdir；失败时类型为 RTempDirectoryUnavailableException
     */
    public CheckResultMsg<String> check(CustomRserveSession session) {
        try {
            Object raw = session.eval(HEALTH_CHECK_COMMAND);
            String response = raw == null ? "" : raw.toString();
            String[] fields = response.split("\\t", -1);
            if (fields.length != 4) {
                return failure("R临时目录健康检查返回格式无效：" + response);
            }

            String tempDirectory = fields[0];
            boolean exists = Boolean.parseBoolean(fields[1]);
            boolean writable = Boolean.parseBoolean(fields[2]);
            boolean probeCreated = Boolean.parseBoolean(fields[3]);
            log.info("R临时目录检查：tempDir={}, exists={}, writable={}, probeCreated={}",
                    tempDirectory, exists, writable, probeCreated);

            if (tempDirectory.isEmpty() || !exists || !writable || !probeCreated) {
                return failure("R临时目录不可用：" + response);
            }

            CheckResultMsg<String> result = new CheckResultMsg<>();
            result.setData(tempDirectory);
            return result;
        } catch (Exception e) {
            log.warn("R临时目录健康检查异常", e);
            return failure("R临时目录健康检查异常：" + e.getMessage());
        }
    }

    private CheckResultMsg<String> failure(String message) {
        CheckResultMsg<String> result = new CheckResultMsg<>(false, message);
        result.setErrorType(RTempDirectoryUnavailableException.class);
        return result;
    }
}

        ROperationParam rOperationParam,

    CheckResultMsg<String> lastFailure = failure(
            RConnectionUnavailableException.class, "R连接失败，请联系软件管理员！");

    for (int attempt = 1; attempt <= MAX_CONNECTION_ATTEMPTS; attempt++) {

                rOperationParam.setRserveSession(session);

                closeAndClear(rOperationParam);
                continue;
            }
        }

        if (isNotConnected(session)) {
            lastFailure = failure(RConnectionUnavailableException.class,
                    "Rserve连接未进入READY状态");
            closeAndClear(rOperationParam);
            continue;
        }

        CheckResultMsg<String> healthResult = healthChecker.check(session);
        if (healthResult.isResult()) {

                    session, rOperationParam.getWorkingRoot()));

                    attempt, healthResult.getData(), rOperationParam.getWorkRoot());

private void handleFailure(CommonTask commonTask,
                           CommonTaskCompletionContext context,
                           Class<?> errorType,
                           Throwable error) {
    Long jobId = context.getJobId();
    String ownerId = context.getOwnerId();
    Long fencingToken = context.getFencingToken();
    AckHandle ackHandle = context.getAckHandle();
    String errorMessage = error != null ? error.getMessage() : "未知错误";
    int retryCount = context.getRetryCount() == null ? 0 : context.getRetryCount();
    boolean retryable = failureClassifier.isRetryable(errorType, error);

    if (retryable && retryCount < retryProperties.getMaxRetries()) {
        boolean marked = jobLeaseService.markRetryableFailure(
                jobId, ownerId, fencingToken, errorMessage);
        if (marked) {
            log.info("任务标记为可重试失败，不通知、不ACK：jobId={}, retryCount={}, maxRetries={}, error={}",
                    jobId, retryCount + 1, retryProperties.getMaxRetries(), errorMessage);
        } else {
            log.warn("标记可重试失败失败（已失权），不通知、不ACK：jobId={}", jobId);
        }
        return;
    }

    boolean marked = jobLeaseService.markFinalFailure(
            jobId, ownerId, fencingToken, errorMessage);
    if (!marked) {
        log.warn("markFinalFailure失败（已失权），不通知、不ACK：jobId={}", jobId);
        return;
    }

    try {
        taskEventPublisher.publishNotificationOrThrow(
                commonTask,
                CommonAnalysisNoticeType.STATUS_CHANGE.getType(),
                CommonSerializableAnalysisStatus.FAIL.getType(),
                true,
                MessageType.ERROR.getType(),
                errorMessage,
                null
        );
        if (ackHandle != null) {
            ackHandle.ack();
        }
        log.info("任务最终失败已处理：jobId={}, retryable={}, retryCount={}, maxRetries={}, error={}",
                jobId, retryable, retryCount, retryProperties.getMaxRetries(), errorMessage);
    } catch (Exception e) {
        log.error("失败通知发布失败，跳过ACK：jobId={}, ownerId={}", jobId, ownerId, e);
    }
}
