# 小程序数据画图功能实施计划

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为微信小程序实现"数据画图"历史记录查看功能，并为开发中的功能添加明确标识，确保符合微信审核规范。

**Tech Stack:** 微信小程序原生框架（WXML/WXSS/JS）、wx.navigateTo、wx.previewImage、wx.saveImageToPhotosAlbum

- [ ] **Step 3: 创建 index.js 页面逻辑**

```javascript
const { chartApi } = require("../../utils/api");

Page({
  data: {
    historyList: [],
    loading: false,
    refreshing: false,
    noMore: false,
    page: 0,
    pageSize: 10,
    totalElements: 0
  },

  onLoad: function () {
    this.loadHistoryList();
  },

  // 加载历史记录列表
  loadHistoryList: function (isRefresh) {
    if (this.data.loading) return;

    if (isRefresh) {
      this.setData({ page: 0, historyList: [], noMore: false });
    }

    this.setData({ loading: true });

    chartApi.getHistoryList({
      page: {
        page: this.data.page,
        pageSize: this.data.pageSize,
        sort: ['id,desc']
      }
    }).then(res => {
      if (res.code === 200) {
        const newList = isRefresh
          ? res.data.content
          : this.data.historyList.concat(res.data.content);
        this.setData({
          historyList: newList,
          totalElements: res.data.totalElements,
          loading: false,
          refreshing: false,
          noMore: newList.length >= res.data.totalElements
        });
      } else {
        this.showError(res.message || '加载失败');
      }
    }).catch(() => {
      this.showError('网络错误，请稍后重试');
    });
  },

  // 下拉刷新
  onPullDownRefresh: function () {
    this.setData({ refreshing: true });
    this.loadHistoryList(true);
  },

  // 上拉加载更多
  onReachBottom: function () {
    if (this.data.noMore || this.data.loading) return;
    this.setData({ page: this.data.page + 1 });
    this.loadHistoryList(false);
  },

  // 跳转详情页
  viewDetail: function (e) {
    const { id, image } = e.currentTarget.dataset;
    wx.navigateTo({
      url: '/pages/chart-detail/index?id=' + id + '&image=' + encodeURIComponent(image)
    });
  },

  // 显示错误提示
  showError: function (message) {
    this.setData({ loading: false, refreshing: false });
    wx.showToast({ title: message, icon: 'none', duration: 2000 });
  }
});
```

```javascript
Page({
  data: {
    id: '',
    imageUrl: ''
  },

  onLoad: function (options) {
    this.setData({
      id: options.id,
      imageUrl: decodeURIComponent(options.image)
    });
  },

  // 全屏预览图片
  previewImage: function () {
    wx.previewImage({
      urls: [this.data.imageUrl],
      current: this.data.imageUrl
    });
  },

  // 保存图片到相册
  saveImage: function () {
    wx.showLoading({ title: '保存中...', mask: true });

    wx.downloadFile({
      url: this.data.imageUrl,
      success: (res) => {
        if (res.statusCode === 200) {
          wx.saveImageToPhotosAlbum({
            filePath: res.tempFilePath,
            success: () => {
              wx.hideLoading();
              wx.showToast({ title: '保存成功', icon: 'success', duration: 2000 });
            },
            fail: (err) => {
              wx.hideLoading();
              if (err.errMsg && err.errMsg.includes('auth deny')) {
                wx.showModal({
                  title: '需要相册权限',
                  content: '请授权访问相册以保存图片',
                  confirmText: '去授权',
                  success: (modalRes) => {
                    if (modalRes.confirm) wx.openSetting();
                  }
                });
              } else {
                wx.showToast({ title: '保存失败', icon: 'none', duration: 2000 });
              }
            }
          });
        } else {
          wx.hideLoading();
          wx.showToast({ title: '下载失败', icon: 'none', duration: 2000 });
        }
      },
      fail: () => {
        wx.hideLoading();
        wx.showToast({ title: '下载失败', icon: 'none', duration: 2000 });
      }
    });
  }
});
```

## 注意事项

1. **后端接口路径**：`chartApi.getHistoryList` 调用 `/asa/stock/analysis/page`，需确认后端已部署该接口
2. **图片域名白名单**：微信小程序需在后台配置 `downloadFile` 合法域名，否则保存图片会失败
3. **Token 鉴权**：`utils/request.js` 的 `buildHeaders` 已自动注入 Bearer Token，无需额外处理
4. **测试账号**：审核时需提供可登录的测试账号，确保审核员能看到历史记录数据
