# M3U8 下载工具 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 `m3u8_downloader` 目录下实现一个可执行的 Node 命令行工具，使用系统 `ffmpeg` 下载 m3u8 并输出 MP4，同时支持用指定测试链接做真实下载验证。

**Architecture:** 工具采用独立 Node CLI 结构，入口层负责解析命令行参数与打印日志，下载层负责组装下载配置与输出路径，`ffmpeg` 适配层负责探测本机依赖并执行实际下载命令。测试分为纯本地单元测试和一个显式触发的真实链接集成验证，确保功能可验证但不把网络依赖混入普通测试。

**Tech Stack:** Node.js 20、`node:test`、`node:assert/strict`、系统 `ffmpeg`

import path from 'node:path';

export function buildDownloadOptions({ url, output, cwd = process.cwd() }) {
  if (!url || !String(url).trim()) {
    throw new Error('m3u8 URL 不能为空');
  }

  const downloadsDir = path.join(cwd, 'downloads');
  const fileName = output && String(output).trim()
    ? withMp4Extension(String(output).trim())
    : `m3u8-download-${Date.now()}.mp4`;

  return {
    url: String(url).trim(),
    downloadsDir,
    outputPath: path.join(downloadsDir, fileName)
  };
}

function withMp4Extension(fileName) {
  return fileName.endsWith('.mp4') ? fileName : `${fileName}.mp4`;
}
```

import { execFile } from 'node:child_process';
import { promisify } from 'node:util';

const execFileAsync = promisify(execFile);

export async function ensureFfmpegAvailable({ runCommand = execFileAsync } = {}) {
  try {
    await runCommand('ffmpeg', ['-version']);
  } catch (error) {
    if (error && error.code === 'ENOENT') {
      throw new Error('未检测到 ffmpeg，请先在系统中安装 ffmpeg');
    }
    throw error;
  }
}

export function buildFfmpegArgs({ url, outputPath }) {
  return ['-y', '-i', url, '-c', 'copy', outputPath];
}

import { mkdir } from 'node:fs/promises';
import { buildDownloadOptions } from './downloader.js';
import { ensureFfmpegAvailable, runFfmpegDownload } from './ffmpeg.js';

export function createDownloader(deps = {}) {
  const ensure = deps.ensureFfmpegAvailable ?? ensureFfmpegAvailable;
  const runDownload = deps.runFfmpegDownload ?? runFfmpegDownload;

  return {
    async download(input) {
      const options = buildDownloadOptions(input);
      await mkdir(options.downloadsDir, { recursive: true });
      await ensure();
      await runDownload(options);
      return options;
    }
  };
}
```

```js
#!/usr/bin/env node

import { createDownloader } from './downloader.js';

export function parseCliArgs(argv) {
  const args = [...argv];
  const result = { url: '', output: '' };

  while (args.length > 0) {
    const current = args.shift();
    if (!result.url && !current.startsWith('--')) {
      result.url = current;
      continue;
    }
    if (current === '--output' || current === '-o') {
      result.output = args.shift() ?? '';
    }
  }

  return result;
}
```
