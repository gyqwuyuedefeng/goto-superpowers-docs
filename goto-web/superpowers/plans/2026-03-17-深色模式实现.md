# 深色模式实现计划

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**目标：** 为 goto-web 项目实现深色模式支持，使页面主题能够自动跟随系统主题，同时提供手动切换功能

**架构：** 使用 CSS 变量 + 媒体查询实现主题切换，通过 Vuex 管理主题状态，localStorage 持久化用户偏好，0.3 秒平滑过渡动画

**技术栈：** Vue 2, Vuex, SCSS, Element UI, localStorage

---

**文件：**

- [ ] **Step 1: 创建 theme.js 工具类**

```javascript
/**
 * 主题管理工具类
 * 负责主题的初始化、切换、持久化和系统主题监听
 */
export const ThemeManager = {
  /**
   * 初始化主题
   * 在应用启动时调用（main.js）
   */
  init() {
    // 从 localStorage 读取保存的主题偏好
    const savedTheme = this.getSavedTheme()

    // 应用主题到 DOM
    this.applyTheme(savedTheme)

    // 启动系统主题变化监听
    this.listenSystemThemeChange()
  },

  /**
   * 获取保存的主题偏好
   * @returns {string} 'auto' | 'light' | 'dark'
   */
  getSavedTheme() {
    try {
      return localStorage.getItem('theme') || 'auto'
    } catch (e) {
      console.warn('无法读取主题偏好:', e)
      return 'auto'
    }
  },

  /**
   * 应用主题到 DOM
   * @param {string} theme - 'auto' | 'light' | 'dark'
   */
  applyTheme(theme) {
    const root = document.documentElement

    if (theme === 'auto') {
      // 移除 data-theme 属性，让媒体查询生效
      root.removeAttribute('data-theme')
    } else {
      // 设置 data-theme 属性
      root.setAttribute('data-theme', theme)
    }

    // 保存到 localStorage
    this.saveTheme(theme)
  },

  /**
   * 保存主题到 localStorage
   * @param {string} theme - 'auto' | 'light' | 'dark'
   */
  saveTheme(theme) {
    try {
      localStorage.setItem('theme', theme)
    } catch (e) {
      console.warn('无法保存主题偏好到 localStorage:', e)
      // 降级：仅使用内存状态，不持久化
    }
  },

  /**
   * 获取当前实际显示的主题（考虑系统偏好）
   * @returns {string} 'light' | 'dark'
   */
  getCurrentTheme() {
    const savedTheme = this.getSavedTheme()

    if (savedTheme === 'auto') {
      // 检测系统主题
      return window.matchMedia('(prefers-color-scheme: dark)').matches
        ? 'dark'
        : 'light'
    }

    return savedTheme
  },

  /**
   * 监听系统主题变化
   */
  listenSystemThemeChange() {
    // 检查浏览器是否支持 matchMedia
    if (!window.matchMedia) {
      console.warn('浏览器不支持 matchMedia API')
      return
    }

    const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)')

    // 监听系统主题变化
    const handleChange = (e) => {
      const savedTheme = this.getSavedTheme()
      // 只有在 auto 模式下才响应系统主题变化
      if (savedTheme === 'auto') {
        console.log('系统主题已变化:', e.matches ? 'dark' : 'light')
        // CSS 会自动响应，无需手动操作
      }
    }

    // 添加事件监听器
    if (mediaQuery.addEventListener) {
      mediaQuery.addEventListener('change', handleChange)
    } else if (mediaQuery.addListener) {
      // 兼容旧版浏览器
      mediaQuery.addListener(handleChange)
    }
  },

  /**
   * 检查浏览器是否支持 CSS 变量
   * @returns {boolean}
   */
  isCSSVariableSupported() {
    return (
      window.CSS &&
      window.CSS.supports &&
      window.CSS.supports('color', 'var(--test)')
    )
  }
}
```

```vue
  <div class="theme-toggle" @click="toggleTheme" :title="themeTooltip">
    <i :class="themeIcon"></i>
  </div>

<script>
export default {
  name: 'ThemeToggle',
  computed: {
    currentTheme() {
      return this.$store.state.settings.theme || 'auto'
    },
    themeIcon() {
      if (this.currentTheme === 'dark') {
        return 'el-icon-moon'
      } else if (this.currentTheme === 'light') {
        return 'el-icon-sunny'
      } else {
        return 'el-icon-monitor'
      }
    },
    themeTooltip() {
      const tooltips = {
        auto: '自动跟随系统',
        light: '浅色模式',
        dark: '深色模式'
      }
      return tooltips[this.currentTheme]
    }
  },
  methods: {
    toggleTheme() {
      const themes = ['auto', 'light', 'dark']
      const currentIndex = themes.indexOf(this.currentTheme)
      const nextTheme = themes[(currentIndex + 1) % themes.length]
      this.$store.dispatch('settings/setTheme', nextTheme)
    }
  }
}
</script>
```
